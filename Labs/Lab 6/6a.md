## Task 6a
```
#include <sys/types.h>
#include <stdio.h>
#include <string.h>
#include <unistd.h>

#define BUFFER_SIZE 25
#define READ_END 0
#define WRITE_END 1

int main(){
    char w_msg[BUFFER_SIZE] = "Greetings";  // message to send through the pipe
    char r_msg[BUFFER_SIZE];  // buffer to read the message into
    int fd[2];  // file descriptors for the pipe
    pid_t pid;

    if (pipe(fd) == -1) {
        printf("Pipe Failed\n");
        return 1;
    }

    pid = fork();

    if (pid < 0) {
        printf("Fork Failed\n");
        return 1;
    } 
    else if (pid > 0) { // Parent process
        close(fd[READ_END]);  // Close the unused read end of the pipe
        write(fd[WRITE_END], w_msg, strlen(w_msg) + 1);  // Write the message into the pipe
        close(fd[WRITE_END]);  // Close the write end after writing
    } 
    else { // Child process
        close(fd[WRITE_END]);  // Close the unused write end of the pipe
        read(fd[READ_END], r_msg, BUFFER_SIZE);  // Read from the pipe
        printf("Read: %s\n", r_msg);  // Output the received message
        printf("Child Process Id: %d\n", getpid());  // Print the child's PID
        close(fd[READ_END]);  // Close the read end after reading
    }

    return 0;
}
```
